% ---------- RENDER HELPERS ----------
% Renders an internship entry (uses InternshipEntry fields)
\newcommand{\renderInternship}[1]{
  \resumeSubheading
    {\csname #1@Company\endcsname}
    {\csname #1@Start\endcsname\ --\ \csname #1@End\endcsname}
    {\csname #1@Role\endcsname}
    {\csname #1@Location\endcsname}
  \csname #1@Bullets\endcsname
}

% Renders a project or hackathon entry
\newcommand{\renderProject}[1]{
  \resumeSubheading
    {\csname #1@Title\endcsname}
    {\csname #1@Date\endcsname}
    {\csname #1@Subtitle\endcsname}{}
  \csname #1@Bullets\endcsname
}
